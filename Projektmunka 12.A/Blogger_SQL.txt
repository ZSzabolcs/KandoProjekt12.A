DROP DATABASE IF EXISTS blogger;

CREATE DATABASE IF NOT EXISTS blogger DEFAULT CHARACTER SET utf8;

CREATE TABLE blogger.user (
username VARCHAR(30) PRIMARY KEY NOT NULL,
email VARCHAR(40) NOT NULL,
password VARCHAR(30) NOT NULL
)
ENGINE = InnoDB;

CREATE TABLE `blogger`.`blog` (
`id` INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
`username` VARCHAR(30) NOT NULL , 
`blog_title` VARCHAR(50) NOT NULL UNIQUE,
`blog_content` TEXT NOT NULL,
`blog_made_date` DATE NOT NULL,
`shared_blog_number` INT DEFAULT 0,
CONSTRAINT FK_blogger_blog FOREIGN KEY (username)
REFERENCES user(username)
)
ENGINE = InnoDB;

CREATE TABLE `blogger`.`comment` (
`id` INT PRIMARY KEY AUTO_INCREMENT NOT NULL, 
`username` VARCHAR(30) NOT NULL,
`blog_title` VARCHAR(50) NOT NULL,
`comment_content` TEXT NOT NULL, 
`comment_date` DATE NOT NULL, 
`like_number` INT DEFAULT 0, 
`dislike_number` INT DEFAULT 0,
CONSTRAINT FK_blog_comment FOREIGN KEY (username)
REFERENCES user(username),
CONSTRAINT FK_blog_title FOREIGN KEY (blog_title)
REFERENCES blog(blog_title)
 )
 ENGINE = InnoDB;

CREATE TABLE `blogger`.`chat`(
`id` INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
`chatroom_name` VARCHAR(30) NOT NULL,
`chat_leader` VARCHAR(30) NOT NULL
)
 ENGINE = InnoDB;
